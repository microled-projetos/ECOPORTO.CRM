(

    CONTAID NUMBER(*,0) NOT NULL,

    VALIDADE DATE, 
    PROTOCOLO VARCHAR2(50 BYTE), 
    TOTALDIVIDA NUMBER, 
    INADIMPLENTE NUMBER(1,0), 
    QUANTIDADE INT,
    LIMITENEGOCIADO NUMBER,
    CREDITOAPROVADO NUMBER(1,0),
    FOREIGN KEY (CONTAID) REFERENCES CRM.TB_CRM_CONTAS (ID) ENABLE
)
